<script>

    // LOAD TENANTS
    fetch("/tenant/all")
        .then(res => res.json())
        .then(data => {
            data.sort((a, b) => a.id - b.id);

            let table = document.getElementById("tenantTable");
            let index = 1;

            data.forEach(t => {
                let row = table.insertRow();

                row.insertCell(0).innerText = index++;
                row.insertCell(1).innerText = t.name;
                row.insertCell(2).innerText = t.phone;
                row.insertCell(3).innerText = t.roomNumber;
                row.insertCell(4).innerText = t.sharingType;
                row.insertCell(5).innerText = t.rentPaid;

                let action = row.insertCell(6);

                let delBtn = document.createElement("button");
                delBtn.innerText = "Delete";
                delBtn.className = "btn delete";
                delBtn.onclick = () => deleteTenant(t.id);

                action.appendChild(delBtn);
            });
        });

    function deleteTenant(id) {
        fetch("/tenant/delete/" + id, {
            method: "DELETE"
        }).then(() => location.reload());
    }

    // LOAD ROOM REQUESTS
    fetch("/request/all")
        .then(res => res.json())
        .then(data => {
            let table = document.getElementById("requestTable");

            data.forEach(r => {
                let row = table.insertRow();

                row.insertCell(0).innerText = r.id;
                row.insertCell(1).innerText = r.roomType;
                row.insertCell(2).innerText = r.status;

                let action = row.insertCell(3);

                let approveBtn = document.createElement("button");
                approveBtn.innerText = "Approve";
                approveBtn.className = "btn approve";
                approveBtn.onclick = () => updateRequest(r.id, "approve");

                let rejectBtn = document.createElement("button");
                rejectBtn.innerText = "Reject";
                rejectBtn.className = "btn reject";
                rejectBtn.onclick = () => updateRequest(r.id, "reject");

                action.appendChild(approveBtn);
                action.appendChild(rejectBtn);
            });
        });

    function updateRequest(id, type) {
        fetch("/request/" + type + "/" + id, {
            method: "PUT"
        }).then(() => location.reload());
    }

    // LOAD PAYMENTS
    fetch("/payment/all")
        .then(res => res.json())
        .then(data => {
            let table = document.getElementById("paymentTable");

            data.forEach(p => {
                let row = table.insertRow();

                row.insertCell(0).innerText = p.id;
                row.insertCell(1).innerText = p.name;
                row.insertCell(2).innerText = p.roomNumber;

                let link = document.createElement("a");
                link.innerText = "View Receipt";
                link.href = "/uploads/" + p.fileName;
                link.target = "_blank";

                row.insertCell(3).appendChild(link);
            });
        });

    // BACK TO HOME
    function goHome() {
        window.location.href = "/index.html";
    }

</script>
